25 template <class T>
26 T box_muller(T m, T s) /* normal random variate generator */
27 { /* mean m, standard deviation s */
28  T x1, x2, w, y1;
29  static T y2;
30  static char use_last = 0;
31 
32  if (use_last) /* use value from previous call */
33  {
34  y1 = y2;
35  use_last = 0;
36  }
37  else
38  {
39  do {
40  x1 = 2.0 * (T)rand()/RAND_MAX - 1.0;
41  x2 = 2.0 * (T)rand()/RAND_MAX - 1.0;
42  w = x1 * x1 + x2 * x2;
43  } while ( w >= 1.0 );
44 
45  w = sqrt( (-2.0 * log( w ) ) / w );
46  y1 = x1 * w;
47  y2 = x2 * w;
48  use_last = 1;
49  }
50 
51  return( m + y1 * s );
52 }
